\begin{song}{title = Boty proti lásce}
\textsc{1.} 
\begin{verse}
\chord{E^7}Zas mi říkal, že má něco pro mě \\
A to něco, to prý láska je. \\
\chord{A^7}Já však nechci žádnou lásku v domě  \\
\chord{E^7}Přináší jen žal a výdaje. \\
\end{verse}

\begin{chorus}
chord{G}Mám proti lásce \chord{E}boty \\
Ty \chord{G}chrání paní \chord{E}svou \\
Ty \chord{G}boty vždycky \chord{E^7}jdou  \\
\chord{(break)}A všechno hezké pošlapou. \\
\end{chorus}

\begin{verse}
\chord{E^7}Žiju sama díky těmto botám \\
žiju sama zásluhou těch bot. \\
\chord{A^7}Často sice klukům hlavy motám \\
\chord{E^7}Botám ale city nejsou vhod \\
\end{verse}

\begin{chorus}
\chord{G}V těch botách ráda \chord{E}šlapu  \\
\chord{G}Zvlášt' po lásce nás \chord{E}dvou \\
\chord{G}Ty boty vždycky \chord{E^7}jdou \\
A všechno hezké pošlapou. \\
\end{chorus}

\begin{verse}
\chord{E^7}Tyhle boty nosím někde v duši \\
Někde v duši, ne však na nohou. \\
\chord{A^7}Jsou tam dole, kde mi srdce buší \\
\chord{E^7}Od bušení mu však pomohou. \\
\\
\end{verse}

\begin{chorus}
\chord{G}V těch botách ráda \chord{E}šlapu \\
\chord{G}A to se mi prý \chord{E}mstí \\
Když \chord{G}prý v těch botách \chord{E^7}jdu \\
A tak si šlapu po štěstí. \\
\end{chorus}

Pozor boty! \\
Pochodem vchod! \\
\end{song}
